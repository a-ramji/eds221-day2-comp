typeof(weights_integer)

Access ranges of values in a vector

Use : to refer to a range (low : high)

trees <- c("ponderosa", "lodgepole", "jeffrey", "sugar", "white", "bristlecone")
# class(trees)

trees[3]
[1] "jeffrey"
trees[3:5]
[1] "jeffrey" "sugar"   "white"
